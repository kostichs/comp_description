# Company Canvas v08 Deployment Commands (With Validation)

# 1. Pull new version with validator
docker pull sergeykostichev/company-canvas-app:v08

# 2. Stop and remove old container
docker stop company-canvas-prod
docker rm company-canvas-prod

# 3. Start new version with validation
docker run -d --name company-canvas-prod -p 80:8000 -v /srv/company-canvas/output:/app/output -e OPENAI_API_KEY=YOUR_OPENAI_KEY -e SERPER_API_KEY=YOUR_SERPER_KEY -e SCRAPINGBEE_API_KEY=YOUR_SCRAPINGBEE_KEY -e HUBSPOT_API_KEY=YOUR_HUBSPOT_KEY sergeykostichev/company-canvas-app:v08

# 4. Check status
docker ps

# 5. Check logs
docker logs company-canvas-prod

# - Detects person names vs company names
